#include <bits/stdc++.h>
#define ll long long
using namespace std;

vector<int> EratoSieve(int n)
{
    vector<int> primes;
    /*关键是合数、素数的概念，
    每个合数都只被最小的因子消去的话，就可以最简单完成筛子。
    埃拉托色尼筛法：把区间内合数筛去，因此只要判断每个数是否是合数再筛去即可。
    正常方法是将素数倍数筛去
    但为了实现线性时间内，开始一个一个数直接判断，这样时间控制在c*n内。
    */
    for (int i = 2; i <= n; i++)
    {
        bool notPrime = false;
        for (auto it = primes.begin(); it != primes.end() && *it < i; it++)
        {
            if (i % *it == 0)
            {
                notPrime = true;
                break;
            }
        }
        if (!notPrime)
            primes.push_back(i);
    }
    return primes;
}

int main()
{
    int n = 100;
    vector<int> tmpe;
    tmpe = EratoSieve(n);
    for (auto it : tmpe)
    {
        cout << it << " ";
    }
    return 0;
}
