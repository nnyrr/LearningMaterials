#include <bits/stdc++.h>
#define ll long long
using namespace std;

vector<int> EratoSieve(int n)
{
    vector<int> primes;
    vector<bool> isPrime(n+1, true);
    isPrime[0] = false, isPrime[1] = true;
    /*关键是合数、素数的概念，
    每个合数都只被最小的因子消去的话，就可以最简单完成筛子。
    埃拉托色尼筛法：把区间内合数筛去，因此只要判断每个数是否是合数再筛去即可。
    将素数倍数筛去
    */
    for(int i=2;i*i<=n;i++){
        if(isPrime[i]){
            for(int j = i*i;j<=n;j+=i){
                isPrime[j]=false;
            }//从i*i开始是因为对于i*k(k<i)，在k的时候就处理过了。
        }
    }
    for(int i=2;i<=n;i++){
        if(isPrime[i]){
            primes.push_back(i);
        }
    }
    return primes;
}

int main()
{
    int n = 100;
    vector<int> tmpe;
    tmpe = EratoSieve(n);
    for (auto it : tmpe)
    {
        cout << it << " ";
    }
    return 0;
}
