#include <bits/stdc++.h>
#define ll long long
using namespace std;

void PrintYHTriangle(int n)
{
    int triMatrix[n][n];
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < n; j++)
        {
            triMatrix[i][j] = 0;
        }
    }
    triMatrix[0][0] = 1;
    int maxVal = 0;
    for (int i = 1; i < n; i++)
    {
        for (int j = 0; j < n; j++)
        {
            triMatrix[i][j] = triMatrix[i - 1][(n + j) % n] + triMatrix[i - 1][(n + j - 1) % n];
            maxVal = max(maxVal, triMatrix[i][j]);
            if (!triMatrix[i][j])
                break;
        }
    }
    int maxDigit = to_string(maxVal).length();
    for (int i = 0; i < n; i++)
    {
        for (int t = 0; t < (n - 1 - i) * maxDigit; t++)
        {
            cout << " ";
        }
        for (int j = 0; j < n && triMatrix[i][j] != 0; j++)
        {
            int curDigit = to_string(triMatrix[i][j]).length();
            int preSpace = (maxDigit - curDigit) / 2;
            int rearSpace = (maxDigit - curDigit) - preSpace;
            for (int k = 0; k < preSpace; k++)
                cout << " ";

            cout << triMatrix[i][j];
            for (int k = 0; k < rearSpace; k++)
                cout << " ";
            for (int k = 0; k < maxDigit; k++)
                cout << " ";
        }
        cout << endl;
    }
    /*
    主要是输出的问题，关于杨辉三角形可以转化为变体，然后每个等于上行同列与前一列之和。
    输出：为了实现三角形结构，需要把所有数字的长度都固定为最长数字的长度，然后两个数字之间（字距）的距离也就是字长（最长数字的长度）。
    然后前导空格也可以通过计算得出。
    */
    return;
}

int main()
{
    int n;
    cin >> n;
    PrintYHTriangle(n);
    return 0;
}
